Joins
Inner join — only matching rows
Use when: "customers who have placed orders", "employees with departments"
SELECT a.name, b.salary
FROM Employees a
JOIN Salaries b ON a.id = b.employee_id
Left join — all left rows, null if no match on right
Use when: "all customers including those with no orders"
SELECT a.name, b.order_id
FROM Customers a
LEFT JOIN Orders b ON a.id = b.customer_id
Find unmatched rows (anti-join)
Use when: "customers with NO orders"
SELECT a.name
FROM Customers a
LEFT JOIN Orders b ON a.id = b.customer_id
WHERE b.customer_id IS NULL
Self join — table joins itself
Use when: "employee and their manager" (both in same table)
SELECT e.name AS employee, m.name AS manager
FROM Employees e
JOIN Employees m ON e.manager_id = m.id
Cross join — every combination
Use when: "tournament schedule", "all possible pairs"
WITH numbered AS (
  SELECT name, ROW_NUMBER() OVER (ORDER BY name) AS id
  FROM Countries
)
SELECT a.name AS team1, b.name AS team2
FROM numbered a
CROSS JOIN numbered b
WHERE a.id < b.id        -- play once each
-- WHERE a.id != b.id   -- home & away (twice)
Left join trap — becomes inner join
Wrong WHERE on right table kills nulls → acts like inner join
-- WRONG: filters out unmatched rows
LEFT JOIN Dept d ON e.dept_id = d.id
WHERE d.name = 'IT'

-- RIGHT: keep condition in ON
LEFT JOIN Dept d ON e.dept_id = d.id AND d.name = 'IT'
Group by / Having / Order by / Limit
Full structure
SELECT department, COUNT(*) AS emp_count, AVG(salary) AS avg_sal
FROM Employees
WHERE salary > 20000          -- filter BEFORE grouping
GROUP BY department
HAVING AVG(salary) > 50000   -- filter AFTER grouping
ORDER BY avg_sal DESC
LIMIT 5
OFFSET 2                      -- skip first 2 rows
Execution order — memorize this
FROM → WHERE → GROUP BY → HAVING → SELECT → ORDER BY → LIMIT → OFFSET
WHERE vs HAVING rule
ScenarioUse
Filter on regular columnWHERE
Filter on COUNT / SUM / AVGHAVING
Both neededWHERE + HAVING together
-- ERROR: aggregates not allowed in WHERE
WHERE COUNT(*) > 5   -- ❌
HAVING COUNT(*) > 5  -- ✅
Group by rule — every non-aggregate in SELECT must be in GROUP BY
SELECT department, name, AVG(salary)
FROM Employees
GROUP BY department, name  -- both non-aggregate columns here
CTE (Common Table Expression)
Single CTE
WITH cte_name AS (
  SELECT col1, AVG(col2) AS avg_val
  FROM table
  GROUP BY col1
)
SELECT col1, avg_val
FROM cte_name
WHERE avg_val > 1000
Chained CTEs — multi table joins
WITH step1 AS (
  SELECT a.id, a.name, b.salary
  FROM Employees a
  JOIN Salaries b ON a.id = b.employee_id
),
step2 AS (
  SELECT s.id, s.name, s.salary, d.dept_name
  FROM step1 s
  JOIN Departments d ON s.id = d.employee_id
)
SELECT * FROM step2

Aggregate then filterAggregate in CTE, filter outside
Window function structure — OVER
The OVER skeleton
FUNCTION() OVER (
  PARTITION BY col     -- reset per group
  ORDER BY col         -- order within group
  ROWS BETWEEN ...     -- only for rolling calcs
)
What each part does
PartWhat it doesOptional?
PARTITION BYSplits into groups, resets function per groupYes
ORDER BYOrders rows within each groupSometimes

Key rule — "for each X" → PARTITION BY X

ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY order_date)

Nth row per group pattern (most common)
WITH ranked AS (
  SELECT col1, col2,
    ROW_NUMBER() OVER (PARTITION BY col1 ORDER BY col2 DESC) AS rn
  FROM table
)
SELECT col1, col2
FROM ranked
WHERE rn = 2   -- change to any N
Top N per group pattern
WITH ranked AS (
  SELECT col1, col2,
    DENSE_RANK() OVER (PARTITION BY col1 ORDER BY col2 DESC) AS rnk
  FROM table
)
SELECT col1, col2
FROM ranked
WHERE rnk <= 3   -- top 3 per group
When to use which
ScenarioUse
"Find Nth order per customer"ROW_NUMBER = N (unique rows)
"Top N products, handle ties"DENSE_RANK <= N
"Rank employees, skip after tie"RANK
LAG / LEAD
Concept
LAG = look at previous row's value. LEAD = look at next row's value. Think of it as standing in a queue — LAG is behind you, LEAD is in front.
Syntax
LAG(column, n, default)  OVER (PARTITION BY ... ORDER BY ...)
LEAD(column, n, default) OVER (PARTITION BY ... ORDER BY ...)
-- n = rows back/forward (default 1)
-- default = what to show instead of NULL (optional)
Week on week change pattern
WITH weekly AS (
  SELECT DATEPART(week, sale_date) AS week_num,
         SUM(revenue) AS weekly_revenue
  FROM Sales
  GROUP BY DATEPART(week, sale_date)
)
SELECT
  week_num,
  weekly_revenue,
  LAG(weekly_revenue, 1) OVER (ORDER BY week_num) AS prev_week,
  weekly_revenue - LAG(weekly_revenue, 1) OVER (ORDER BY week_num) AS wow_change
FROM weekly

ScenarioResult
First row with LAGNULL (no previous row)
Last row with LEADNULL (no next row)

No PARTITION BYLooks across all rows (usually wrong)
Rolling totals / Moving averages
ROWS BETWEEN reference
KeywordMeaning
UNBOUNDED PRECEDINGFrom the very first row
N PRECEDINGN rows back from current
CURRENT ROWThis row
N FOLLOWINGN rows forward
UNBOUNDED FOLLOWINGUntil the last row
Running total — from start to current row
SELECT col1, col2,
  SUM(col2) OVER (
    PARTITION BY col1
    ORDER BY date_col
    ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
  ) AS running_total
FROM table
Moving average — last 3 rows
SELECT col1, col2,
  AVG(col2) OVER (
    PARTITION BY col1
    ORDER BY date_col
    ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
  ) AS moving_avg_3
FROM table

CASE WHEN
Basic structure
SELECT col1,
  CASE
    WHEN col2 IS NULL THEN 'Unknown'   -- NULL check first always
    WHEN col2 >= 90 THEN 'A'
    WHEN col2 BETWEEN 75 AND 89 THEN 'B'
    WHEN col2 BETWEEN 60 AND 74 THEN 'C'
    ELSE 'Fail'
  END AS grade
FROM Students
Conditional aggregation — CASE inside COUNT/SUM
Pivot row values into columns without a separate query
SELECT department,
  COUNT(CASE WHEN salary > 50000 THEN 1 END) AS high_earners,
  COUNT(CASE WHEN salary <= 50000 THEN 1 END) AS low_earners,
  SUM(CASE WHEN gender = 'M' THEN salary ELSE 0 END) AS male_total
FROM Employees
GROUP BY department
CASE inside window function
SELECT page_id, visit_date, visitors,
  CASE
    WHEN LEAD(visitors,1) OVER (PARTITION BY page_id ORDER BY visit_date)
         > visitors THEN 'Increasing'
    ELSE 'Dropping/Same'
  END AS trend
FROM WebTraffic
Rules
RuleDetail
Always close with ENDForgetting END is the most common syntax error
Always give aliasAS grade, AS category etc.
NULL check firstWHEN col IS NULL must be the first WHEN
Stops at first trueSQL checks top to bottom, stops at first match
NULL handling / COALESCE
NULL rules — memorize these
NULL = NULL    -- returns NULL, not True!
NULL != NULL   -- returns NULL
NULL + 5       -- returns NULL
NULL > 5       -- returns NULL

-- NEVER use:
WHERE col = NULL    -- ❌ always returns nothing
WHERE col != NULL   -- ❌ always returns nothing

-- ALWAYS use:
WHERE col IS NULL       -- ✅
WHERE col IS NOT NULL   -- ✅
COALESCE — replace NULL with default
-- Replace NULL with 0 (prevents broken math)
SELECT salary + COALESCE(bonus, 0) AS total
FROM Employees

-- Display replacement
SELECT COALESCE(middle_name, 'N/A') AS middle_name

-- Multiple fallbacks — returns first non-NULL
SELECT COALESCE(phone, mobile, email, 'No contact') AS contact
COUNT(*) vs COUNT(column)
COUNT(*)        -- counts ALL rows including NULLs
COUNT(salary)   -- counts only non-NULL salary rows

-- Find number of NULLs in a column:
SELECT COUNT(*) - COUNT(salary) AS null_count
FROM Employees

Structure
SELECT col1, col2 FROM table1
UNION        -- combines, removes duplicates
UNION ALL    -- combines, keeps duplicates
INTERSECT    -- only rows in BOTH tables
EXCEPT       -- rows in first but NOT in second
SELECT col1, col2 FROM table2

Matching data typesColumn types must be compatible
Column names from firstAlias in first SELECT sets the output column name

Year on year pattern
WITH yearly AS (
  SELECT YEAR(sale_date) AS yr, SUM(revenue) AS total
  FROM Sales
  GROUP BY YEAR(sale_date)
)
SELECT yr, total,
  LAG(total, 1) OVER (ORDER BY yr) AS prev_year,
  total - LAG(total, 1) OVER (ORDER BY yr) AS yoy_change
FROM yearly
BETWEEN date trap
-- Safe for DATE columns
WHERE order_date BETWEEN '2024-01-01' AND '2024-12-31'

-- DATETIME columns — BETWEEN misses Dec 31 after midnight!
-- Use this instead:
WHERE order_date >= '2024-01-01' AND order_date < '2025-01-01'
Tricky MCQ behaviors
Integer division
SELECT 10/3     -- returns 3, NOT 3.33!
SELECT 10.0/3   -- returns 3.333...
NULL comparisons — all return NULL
SELECT NULL = NULL   -- NULL (not True)
SELECT NULL != NULL  -- NULL
SELECT NULL + 5      -- NULL

WHERE salary != NULL   -- returns ZERO rows
WHERE salary = NULL    -- returns ZERO rows
WHERE vs HAVING error
-- ERROR: aggregate in WHERE
SELECT dept, COUNT(*) FROM Employees
WHERE COUNT(*) > 5         -- ❌ throws error
GROUP BY dept

-- Correct
GROUP BY dept
HAVING COUNT(*) > 5        -- ✅
UNION removes duplicates
SELECT 1,2,3 UNION SELECT 1,2,3 UNION SELECT 1,2,3
-- returns 1 row only

SELECT 1,2,3 UNION ALL SELECT 1,2,3 UNION ALL SELECT 1,2,3
-- returns 3 rows
LIMIT vs DENSE_RANK
LIMIT 1            -- exactly 1 row, ignores ties
WHERE rnk = 1       -- all tied rows at rank 1
Left join trap with WHERE
LEFT JOIN d ON e.id = d.id
WHERE d.name = 'IT'   -- ❌ acts like INNER JOIN

LEFT JOIN d ON e.id = d.id AND d.name = 'IT'  -- ✅ true LEFT JOIN
GROUP BY wrong column
SELECT dept, AVG(salary)
FROM Employees
GROUP BY dept, salary   -- ❌ AVG becomes pointless
GROUP BY dept           -- ✅ correct
